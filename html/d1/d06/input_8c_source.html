79bool input_handleByte(const uint8_t *byte, GameState *game_state, CellState *cell_state)
80{
81 switch(*byte)
82 {
83 case '\e':
84 if(!g_timer.is_running)
85 break;
86 handleEscapeSequence(cell_state);
87 break;
88 case '\r':
89 case ' ':
90 handleSetMark(game_state, cell_state);
91 break;
92 case '?':
93 openControls();
94 return true;
95 case 'p':
96 case 'P':
97 return handlePauseResume(cell_state);
98 case '+':
99 cell_increaseSize();
100 return true;
101 case '-':
102 cell_decreaseSize();
103 return true;
104 default:;
105 }
106 return false;
107}

109static void handleEscapeSequence(CellState *cell_state)
110{
111 uint8_t input = '\0';
112 if(!input_getNext(&g_input_buf, &input) || input != '[')
113 {
114 return;
115 }
116
117 if(!input_getNext(&g_input_buf, &input))
118 {
119 return;
120 }
121
122 const int col = cell_state->selected->col;
123 const int row = cell_state->selected->row;
124
125 switch(input)
126 {
127 case 'A':
128 if(row - 1 >= 0)
129 {
130 cell_state->selected = &cell_state->all[row - 1][col];
131 }
132 break;
133 case 'B':
134 if(row + 1 < CELLS_PER_COL)
135 {
136 cell_state->selected = &cell_state->all[row + 1][col];
137 }
138 break;
139 case 'C':
140 if(col + 1 < CELLS_PER_ROW)
141 {
142 cell_state->selected = &cell_state->all[row][col + 1];
143 }
144 break;
145 case 'D':
146 if(col - 1 >= 0)
147 {
148 cell_state->selected = &cell_state->all[row][col - 1];
149 }
150 break;
151 default:;
152 }
153}

155static void handleSetMark(GameState *game_state, CellState *cell_state)
156{
157 if(cell_state->selected->marked_by != None || !g_timer.is_running)
158 {
159 return;
160 }
161
162 if(game_state->mode == PVE)
163 {
164 cell_state->selected->marked_by = Cross;
165 cell_state->last_cross = cell_state->selected;
166
167
168 // Avoid a bug where the bot unfairly wins, if the player started
169 // and both the player and the bot could win in the same round
170 bool hasWinner = false;
171 if(game_state->fields_marked >= (min(CELLS_PER_COL, CELLS_PER_ROW)))
172 {
173 static Cell winningCells[max(CELLS_PER_COL, CELLS_PER_ROW)];
174 if(winning_checkForWinner(cell_state, (Cell **)&winningCells) != None)
175 {
176 hasWinner = true;
177 }
178 }
179
180 if(!hasWinner)
181 {
182 if(game_state->fields_marked + 1 < CELLS_PER_ROW * CELLS_PER_COL)
183 {
184 cell_state->last_circle = bot_markRandomCell(cell_state->all, Circle);
185 }
186 else
187 {
188 game_state->fields_marked--;
189 }
190 }
191
192 cell_redraw(cell_state->last_circle);
193 }
194 else
195 {
196 cell_state->selected->marked_by = game_state->current_player;
197 if(game_state->current_player == Circle)
198 {
199 cell_state->last_circle = cell_state->selected;
200 cell_select(cell_state->last_circle);
201 }
202 else
203 {
204 cell_state->last_cross = cell_state->selected;
205 cell_select(cell_state->last_cross);
206 }
207 }
208
209 game_endTurn();
210 ui_updateTurn(game_state->round, game_state->current_player);
211}
