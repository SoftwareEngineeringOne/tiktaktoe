38bool input_handleByte(uint8_t *byte, GameState *game_state, CellState *cell_state)
39{
40 switch(*byte)
41 {
42 case '\e':
43 if(!g_timer.is_running)
44 break;
45 handleEscapeSequence(cell_state);
46 break;
47 case '\r':
48 case ' ':
49 handleSetMark(game_state, cell_state);
50 break;
51 case '?':
52 openControls();
53 return true;
54 case 'p':
55 case 'P':
56 return handlePauseResume(cell_state);
57 case '+':
58 cell_increaseSize();
59 return true;
60 case '-':
61 cell_decreaseSize();
62 return true;
63 default:;
64 }
65 return false;
66}

68void handleEscapeSequence(CellState *cell_state)
69{
70 uint8_t input = '\0';
71 if(!input_getNext(&g_input_buf, &input) || input != '[')
72 {
73 return;
74 }
75
76 if(!input_getNext(&g_input_buf, &input))
77 {
78 return;
79 }
80
81 const int col = cell_state->selected->col;
82 const int row = cell_state->selected->row;
83
84 switch(input)
85 {
86 case 'A':
87 if(row - 1 >= 0)
88 {
89 cell_state->selected = &cell_state->all[row - 1][col];
90 }
91 break;
92 case 'B':
93 if(row + 1 < CELLS_PER_COL)
94 {
95 cell_state->selected = &cell_state->all[row + 1][col];
96 }
97 break;
98 case 'C':
99 if(col + 1 < CELLS_PER_ROW)
100 {
101 cell_state->selected = &cell_state->all[row][col + 1];
102 }
103 break;
104 case 'D':
105 if(col - 1 >= 0)
106 {
107 cell_state->selected = &cell_state->all[row][col - 1];
108 }
109 break;
110 default:;
111 }
112}

114static void handleSetMark(GameState *game_state, CellState *cell_state)
115{
116 if(cell_state->selected->marked_by != None || !g_timer.is_running)
117 {
118 return;
119 }
120
121 if(game_state->mode == PVE)
122 {
123 cell_state->selected->marked_by = Cross;
124 cell_state->last_cross = cell_state->selected;
125
126
127 // Avoid a bug where the bot unfairly wins, if the player started
128 // and both the player and the bot could win in the same round
129 bool hasWinner = false;
130 if(game_state->fields_marked >= (min(CELLS_PER_COL, CELLS_PER_ROW)))
131 {
132 static Cell winningCells[max(CELLS_PER_COL, CELLS_PER_ROW)];
133 if(winning_checkForWinner(cell_state, (Cell **)&winningCells) != None)
134 {
135 hasWinner = true;
136 }
137 }
138
139 if(!hasWinner)
140 {
141 if(game_state->fields_marked + 1 < CELLS_PER_ROW * CELLS_PER_COL)
142 {
143 cell_state->last_circle = bot_markRandomCell(cell_state->all, Circle);
144 }
145 else
146 {
147 game_state->fields_marked--;
148 }
149 }
150
151 cell_redraw(cell_state->last_circle);
152 }
153 else
154 {
155 cell_state->selected->marked_by = game_state->current_player;
156 if(game_state->current_player == Circle)
157 {
158 cell_state->last_circle = cell_state->selected;
159 cell_select(cell_state->last_circle);
160 }
161 else
162 {
163 cell_state->last_cross = cell_state->selected;
164 cell_select(cell_state->last_cross);
165 }
166 }
167
168 game_endTurn();
169 ui_updateTurn(game_state->round, game_state->current_player);
170}
