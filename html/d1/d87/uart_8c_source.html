81uint8_t uart_readByte()
82{
83 // if not ready, return 0
84 const uint32_t receiveIsReady = register_read(UART_BASE_ADDRESS + UART_RXDRDY);
85
86 if(!receiveIsReady)
87 {
88 // FIFO is not ready to read,
89 // so return 0 instead
90 return 0;
91 }
92
93 // we have to CLEAR the event before reading out from RXD
94 register_write(UART_BASE_ADDRESS + UART_RXDRDY, UART_EVENT_CLEAR);
95
96 // FIFO is ready to read something out of it
97 return register_read(UART_BASE_ADDRESS + UART_RXD);
98}
