64void game_run(const Mode mode)
65{
66 init(mode);
67
68 Cell *winner_cells[max(CELLS_PER_COL, CELLS_PER_ROW)] = {0};// NOLINT
69 uint8_t input;
70
71 if(game_state.mode == PVE && game_state.current_player == Circle)
72 {
73 cell_state.last_circle = bot_markRandomCell(cell_state.all, Circle);
74 game_state.current_player = Cross;
75 game_state.fields_marked++;
76
77 cell_select(cell_state.last_circle);
78 cell_select(&cell_state.all[0][0]);
79 ui_updateTurn(game_state.round, game_state.current_player);
80 }
81
82 while(true)
83 {

86 // If true, Move was forced due to inactivity

88 {
89 handleForcedMoveUpdate();
90 force_ui_update = false;
91 input = 0;
92 break;
93 }
94 ui_updateTimer(REMAINING_TIME, TICKS_PER_TURN);
95 __WFI();

97
98 const bool should_redraw_field = input_handleByte(&input, &game_state, &cell_state);
99
100 game_state.winner = winning_checkForWinner(&cell_state, winner_cells);
101
102 if(g_timer.is_running)
103 {
104 cell_select(cell_state.selected);
105 }
106
107 if(game_state.winner != None || input == 'q' || input == 'Q'
108 || game_state.fields_marked >= CELLS_PER_COL * CELLS_PER_ROW)
109 {
110 timer_stop(TIMER0);
111 break;
112 }
113
114 if(should_redraw_field)
115 {
116 redrawField();
117 }
118 }
119
120 cell_redrawAll(cell_state.all, "\e[0;2m");
121 for(uint8_t i = 0; i < (max(CELLS_PER_COL, CELLS_PER_ROW)) && winner_cells[i] != NULL;
122 i++)// NOLINT
123 {
124 cell_redraw_withModifier(winner_cells[i], BOLD);
125 }
126
127 calculateSummary();
128 menu_showGameOver(&game_state);
129}

228static void calculateSummary()
229{
230 game_state.total_ticks = g_timer.ticks_total;
231
232 if(game_state.mode == PVE)
233 {
234 game_state.average_ticks = g_timer.ticks_total / 2 / game_state.round;
235 }
236 else
237 {
238 game_state.average_ticks = g_timer.ticks_total / game_state.round;
239 }
240
241 // the loser made one turn less
242 uint8_t cross_rounds = game_state.round;
243 uint8_t circle_rounds = game_state.round;
244 if(game_state.winner == Cross)
245 {
246 circle_rounds--;
247 }
248 else if(game_state.winner == Circle)
249 {
250 cross_rounds--;
251 }
252
253 game_state.cross_average_ticks = game_state.cross_total_ticks / cross_rounds;
254 game_state.circle_average_ticks = game_state.circle_total_ticks / circle_rounds;
255}
