33// Register Offsets (from Table 275, page 154)
34// Tasks:
35#define UART_STARTRX 0x000///< Start UART receiver.
36#define UART_STOPRX 0x004 ///< Stop UART receiver.
37#define UART_STARTTX 0x008///< Start UART transmitter.
38#define UART_STOPTX 0x00C ///< Stop UART transmitter.
39#define UART_SUSPEND 0x01C///< Suspend UART.
40
41// Events:
42#define UART_CTS 0x100 ///< CTS is activated (set low). Clear To Send.
43#define UART_NCTS 0x104 ///< CTS is deactivated (set high). Not Clear To Send.
44#define UART_RXDRDY 0x108///< Data received in RXD.
45#define UART_TXDRDY 0x11C///< Data sent from TXD.
46#define UART_ERROR 0x124 ///< Error detected.
47#define UART_RXTO 0x144 ///< Receiver timeout.
48
49// Registers:
50#define UART_INTEN 0x300 ///< Enable or disable interrupt.
51#define UART_INTENSET 0x304///< Enable interrupt.
52#define UART_INTENCLR 0x308///< Disable interrupt.
53#define UART_ERRORSRC 0x480///< Error source.
54#define UART_ENABLE 0x500 ///< Enable UART.
55#define UART_PSELRTS 0x508 ///< Pin select for RTS.
56#define UART_PSELTXD 0x50C ///< Pin select for TXD.
57#define UART_PSELCTS 0x510 ///< Pin select for CTS.
58#define UART_PSELRXD 0x514 ///< Pin select for RXD.
59#define UART_RXD 0x518 ///< RXD register.
60#define UART_TXD 0x51C ///< TXD register.
61#define UART_BAUDRATE 0x524///< Baud rate.
62#define UART_CONFIG 0x56C ///< Configuration of parity and hardware flow control.
63
64// Other:
65#define UART_TASK_START 1 ///< Value for starting a Task.
66#define UART_EVENT_TRIGGER 1///< Value for setting/starting an Event.
67#define UART_EVENT_CLEAR 0 ///< Value for clearing/stopping an Event.
68
69#define UART_INT_RXDRDY (1 << 2)///< Interrupt bit for RXDRDY event.
