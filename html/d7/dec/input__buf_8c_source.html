41void input_onInterrupt(const uint8_t input)
42{
43 if(!putIntoBuf(&g_input_buf, input))
44 {
45 // Triggering a hardfault is not the best way to handle this, but it is
46 // better than silently dropping input.
47 __asm volatile("udf #0");
48 }
49}

67bool input_getNext(volatile InputBuffer *ib, uint8_t *byte)
68{
69 if(input_isEmpty(ib))
70 {
71 return false;
72 }
73 *byte = ib->data[ib->tail];

78static bool putIntoBuf(volatile InputBuffer *ib, const uint8_t byte)
79{
80 if(input_isFull(ib))
81 {
82 return false;
83 }
84
85 ib->data[ib->head] = byte;
