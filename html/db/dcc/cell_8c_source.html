146void redrawCell(const uint8_t row, const uint8_t col, const char *modifier, const Player marked_by)
147{
148 const uint8_t top = row * (g_cell_size.height - 1) + 1;
149 const uint8_t left = col * g_cell_size.width + 1;
150 const uint8_t bot = top + (g_cell_size.height - 2);
151 const uint8_t right = left + (g_cell_size.width - 2);
152
153 for(uint8_t i = top; i < bot; i++)
154 {
155 cursor_moveTo(left + FIELD_X_OFFSET, i + FIELD_Y_OFFSET);
156
157 print(modifier);
158 for(uint8_t j = 0; j < g_cell_size.width - 1; j++)
159 {
160 switch(marked_by)
161 {
162 case Cross:
163 printStyle(&DEFAULT_CROSS);
164 print(modifier);
165 switch(g_cell_size.size)
166 {
167 case Small:
168 print(SMALL_X[i - top][j]);
169 break;
170 case Medium:
171 print(MEDIUM_X[i - top][j]);
172 break;
173 case Large:
174 print(LARGE_X[i - top][j]);
175 break;
176 }
177 break;
178 case Circle:
179 printStyle(&DEFAULT_CIRCLE);
180 print(modifier);
181 switch(g_cell_size.size)
182 {
183 case Small:
184 print(SMALL_O[i - top][j]);
185 break;
186 case Medium:
187 print(MEDIUM_O[i - top][j]);
188 break;
189 case Large:
190 print(LARGE_O[i - top][j]);
191 break;
192 }
193 break;
194 case None:
195 print(modifier);
196 print(" ");
197 break;
198 }
199 }
200
201
202 cursor_moveTo(right + FIELD_X_OFFSET, i + FIELD_Y_OFFSET);
203 print(RESET);
204 }
205}
