34void ui_updateTimer(const uint16_t remaining_time, const uint16_t total_time)
35{
36 cursor_moveTo(2, TIMER_ROW);
37 print(ERASE_LINE_FROM_CURSOR);
38 print(BOLD);
39
40 #ifndef DISABLE_TIMER
41 static Style style;
42 const uint8_t percent = ((float)remaining_time / (float)total_time) * 100;// NOLINT
43 if(!g_timer.is_running)
44 {
45 style = (Style){.font_style = DIM};
46 }
47 else if(percent > 66)
48 {
49 style = (Style){.fg_color = FG_GREEN};
50 }
51 else if(percent > 33)
52 {
53 style = (Style){.fg_color = FG_YELLOW};
54 }
55 else
56 {
57 style = (Style){.fg_color = FG_RED};
58 }
59
60 for(uint16_t i = 0; i < total_time; i++)
61 {
62 if(i < remaining_time)
63 {
64 print_styled(FULL_PROGRESS_ELEMENT, &style);
65 }
66 else if(i == remaining_time)
67 {
68 print_styled(HALF_PROGRESS_ELEMENT, &style);
69 }
70 else
71 {
72 print_styled(EMPTY_PROGRESS_ELEMENT, &style);
73 }
74 }
75
76 cursor_moveTo(2, TIMER_ROW + 1);
77 if(!g_timer.is_running)
78 {
79 println_styled("Paused...", &ITALIC_DIM);
80 return;
81 }
82 print(ERASE_LINE_FROM_CURSOR);
83 #else
84 print_styled("Timer disabled", &ITALIC_DIM);
85 #endif
86 uart_writeByte('\n');
87}

133void ui_printStartingPlayer(Player starting_player, Mode mode)
134{
135 print_clearConsole();
136 ui_printHeading();
137 switch(starting_player)
138 {
139 case Cross:
140 if(mode == PVE)
141 {
142 print_styled(PLAYER_STARTS, &DEFAULT_CROSS);
143 }
144 else
145 {
146 print_styled(CROSS_STARTS, &DEFAULT_CROSS);
147 }
148 break;
149 case Circle:
150 if(mode == PVE)
151 {
152 print_styled(COMPUTER_STARTS, &DEFAULT_CIRCLE);
153 }
154 else
155 {
156 print_styled(CIRCLE_STARTS, &DEFAULT_CIRCLE);
157 }
158 break;
159 case None:
160 print_styled(
161 "Something went very wrong...",
162 &(Style){.font_style = BOLD, .fg_color = FG_RED});
163 }
164 menu_anyKeyToContinue("Press any key to start the game...");
165}
