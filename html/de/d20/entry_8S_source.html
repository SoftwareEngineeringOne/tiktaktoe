92 .global Reset_Handler
93 .type Reset_Handler, %function
94/**
95 * @brief Reset handler.
96 *
97 * @details
98 * This function is the entry point after a reset. It initializes the `.bss`
99 * and `.data` sections and then calls the `main` function.
100 */
101Reset_Handler:
102 bl zeroBss
103 bl copyData
104 bl main
1051:
106 bkpt
107 b 1b

112/**
113 * @brief Zeroes the `.bss` section.
114 *
115 * @details
116 * This function sets all bytes in the `.bss` section to zero.

118zeroBss:

131 .global copyData
132 .type copyData, %function
133/**
134 * @brief Copies the `.data` section from flash to RAM.
135 *
136 * @details
137 * This function initializes the `.data` section by copying its contents
138 * from the load address in flash to the execution address in RAM.
139 */
140copyData:
141 ldr r0, =__data_start__
142 ldr r1, =__data_end__
143 ldr r2, =__data_load_addr__
144 cmp r0, r1
145 bhs 2f
1461:
147 ldr r3, [r2]
148 str r3, [r0]
149 adds r0, #4
150 adds r2, #4
151 cmp r0, r1

1532:
