40void timer_init(Timer const timer)

48void timer_init_detailed(Timer const timer, uint8_t const prescaler, TimerBitMode const bitMode)
49{
50 const uint32_t timerBase = TimerBase[timer];
51 // Instance[0] of the Timer uses Peripheral ID: 8
52
53 // Prescaler
54 register_write(timerBase | TIMER_PRESCALER, prescaler);
55
56 // BitMode
57 register_write(timerBase | TIMER_BITMODE, bitMode);
58
59 // Enable Interrupt
60 register_write(TIMER0_BASE_ADDRESS | TIMER_INTENSET, INT_COMPARE0);// Interrupt on Compare[0]
61
62 // Enable User-Interrupt from Cortex-M0
63 // ID8 ist der Timer0
64 register_write(Interrupt_Set_Enable, Interrupt_ID8);
65}

67void timer_captureCompareSet(
68 const Timer timer,
69 const TimerCaptureCompare captureCompare,
70 const uint32_t value,
71 const bool shortcutClear)
72{
73 const uint32_t timerBase = TimerBase[timer];
74 const uint32_t ccAddress = timerBase | TimerCCOffset[captureCompare];
75
76 register_write(ccAddress, value);
77
78 // Enable Shortcut between CC[0] and CLEAR
79 uint32_t oldValue = register_read(timerBase | TIMER_SHORTS);
80 oldValue &= ~(1 << captureCompare);
81 register_write(timerBase | TIMER_SHORTS, oldValue | (shortcutClear ? 1 : 0) << captureCompare);
82}
