52void timer_init_detailed(
53 Timer const timer,
54 uint8_t const prescaler,
55 TimerMode const mode,
56 TimerBitMode const bitMode)
57{
58 const uint32_t timerBase = TimerBase[timer];
59 // Instance[0] of the Timer uses Peripheral ID: 8
60
61 // Prescaler

63
64 // BitMode

75void timer_captureCompareSet(
76 const Timer timer,
77 const TimerCaptureCompare captureCompare,
78 const uint32_t value,
79 const bool shortcutClear)
80{
81 const uint32_t timerBase = TimerBase[timer];
82 const uint32_t ccAddress = timerBase | TimerCCOffset[captureCompare];
83
84 register_write(ccAddress, value);
85
86 // Enable Shortcut between CC[0] and CLEAR
87 uint32_t oldValue = register_read(timerBase | TIMER_SHORTS);
88 oldValue &= ~(1 << captureCompare);
89 register_write(timerBase | TIMER_SHORTS, oldValue | (shortcutClear ? 1 : 0) << captureCompare);
90}
