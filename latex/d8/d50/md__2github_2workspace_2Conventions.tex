\item Private functions should be marked as {\ttfamily static} and prefixed with "{}\`{}\+\_\+\mbox{[}...\mbox{]}\`{}"{}

\item Other macros should use {\ttfamily SCREAMING\+\_\+\+CASE} naming
\end{DoxyItemize}\hypertarget{md__2github_2workspace_2Conventions_autotoc_md28}{}\doxysection{\texorpdfstring{üìù Documentation}{üìù Documentation}}\label{md__2github_2workspace_2Conventions_autotoc_md28}

\begin{DoxyItemize}
\item All files should contain a top level comment.
\item All public functions should be documented in the header file.
\item All private functions should be documented at the definition in the source files.
\item All structs and enums should be documented.
\item All global variables should be documented.
\item All "{}complicated"{} macros should be documented.
\item All typedefs should be documented.
\end{DoxyItemize}\hypertarget{md__2github_2workspace_2Conventions_autotoc_md29}{}\doxysubsection{\texorpdfstring{Documentation templates}{Documentation templates}}\label{md__2github_2workspace_2Conventions_autotoc_md29}
